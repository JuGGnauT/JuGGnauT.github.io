接口重试函数
2022年12月05日 343 字 大概 1 分钟 
//TODO: 1. 实现一个，有错误机制，重试机制，加载机制的请求函数
//      2. 先做一个，仅跟接口请求有关的

defineAsyncRequest({
    // 加载Promise
    loader: () => new Promise((resolve, reject) => {
        setTimeout(() => {
            Math.random() > .9 ? resolve("请求成功！") : reject("请求失败！")
        }, 3000)
    }),
    // 加载延时
    delay: 1000,
    // 加载中状态变更回调
    onLoading: (state) => {
        console.log(state ? '加载中。。' : '加载完毕')
    },
    // 出错回调，可以重试
    onError: (retry, fail, retries) => {
        console.log(retries)
        if (retries > 3) {
            console.log('失败超过三次，直接当作失败')
            fail()
        } else {
            retry()
        }
    }
})
    .then(res => {
        console.log(`成功 res:${res}`)
    })
    .catch((err) => {
        console.log('err=>', err)
    })


function defineAsyncRequest(options) {
    console.time("开始执行 defineAsyncRequest")
    if (typeof options === 'function') {
        options = {
            loader: options
        }
    }
    const { loader } = options

    // 记录重试次数
    let retries = 0
    let loading = false
    let loaded = false

    function loadingCb(loading) {
        options.onLoading && options.onLoading(loading)
    }

    let loadingTimer = null
    if (options.delay) {
        loadingTimer = setTimeout(() => {
            loading = true
            loadingCb(loading)
        }, options.delay);
    } else {
        loading = true
        loadingCb(loading)
    }

    function load() {
        return loader()
            .catch((err) => {
                if (options.onError) {
                    return new Promise((resolve, reject) => {
                        // 重试
                        const retry = () => {
                            console.log("请求重试----")
                            resolve(load())
                            retries++
                        }
                        const fail = () => reject(err)

                        options.onError(retry, fail, retries)
                    })
                } else {
                    throw err;
                }
            })
    }


    return load()
        .then(c => {
            loaded = true
            return c
        })
        .finally(() => {
            loading = false
            loadingCb(loading)
            clearTimeout(loadingTimer)
        })

}
